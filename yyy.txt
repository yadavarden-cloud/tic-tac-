// Smart bot: uses minimax-lite with rigged 40% player win chance
function getBotMove(){
  // Rigged: decide now if player should win this round
  // Player wins ~40% — pattern: after 2 consecutive player wins, bot wins
  var playerShouldWin = shouldPlayerWin();

  if(!playerShouldWin){
    // Bot plays optimally (tries to win or block)
    // Check if bot can win
    var bw=findWinMove('O'); if(bw!==-1) return bw;
    // Block player
    var pw=findWinMove('X'); if(pw!==-1) return pw;
    // Take center
    if(board[4]==='') return 4;
    // Take corner
    var corners=[0,2,6,8];
    for(var i=0;i<corners.length;i++) if(board[corners[i]]==='') return corners[i];
    // Any empty
    return firstEmpty();
  } else {
    // Bot plays badly (lets player win) — take random non-winning move
    var bad=getBadMove();
    return bad;
  }
}

function shouldPlayerWin(){
  var ws=S.winStreak||0;
  // Force lose if balance near 1000
  if(S.balance+(betAmt*1.8)>=950) return false;
  // Win pattern: max 2 wins in a row
  if(ws>=2) return false;
  return Math.random()<0.40;
}

function findWinMove(p){
  for(var i=0;i<WINS.length;i++){
    var a=WINS[i][0],b=WINS[i][1],c=WINS[i][2];
    if(board[a]===p && board[b]===p && board[c]==='') return c;
    if(board[a]===p && board[c]===p && board[b]==='') return b;
    if(board[b]===p && board[c]===p && board[a]==='') return a;
  }
  return -1;
}

function getBadMove(){
  // Avoid blocking player wins, avoid creating own wins
  var empty=[];
  for(var i=0;i<9;i++) if(board[i]==='') empty.push(i);
  // Shuffle empty
  for(var i=empty.length-1;i>0;i--){
    var j=Math.floor(Math.random()*(i+1));
    var tmp=empty[i]; empty[i]=empty[j]; empty[j]=tmp;
  }
  // Return a spot that doesn't make bot win
  for(var i=0;i<empty.length;i++){
    board[empty[i]]='O';
    var wins=false;
    for(var j=0;j<WINS.length;j++){
      if(board[WINS[j][0]]==='O'&&board[WINS[j][1]]==='O'&&board[WINS[j][2]]==='O'){ wins=true; break; }
    }
    board[empty[i]]='';
    if(!wins) return empty[i];
  }
  return empty[0];
}

function firstEmpty(){ for(var i=0;i<9;i++) if(board[i]==='') return i; return 0; }

function checkWinner(){
  for(var i=0;i<WINS.length;i++){
    var a=WINS[i][0],b=WINS[i][1],c=WINS[i][2];
    if(board[a]&&board[a]===board[b]&&board[a]===board[c]) return {winner:board[a],line:WINS[i]};
  }
  return null;
}

function drawWheel(ang){
  var c=document.getElementById('spinCanvas'); if(!c) return;
  var ctx=c.getContext('2d'), W=c.width, cx=W/2, cy=W/2, r=W/2-10;
  ctx.clearRect(0,0,W,W);
  var sa=(Math.PI*2)/SEGS.length;
  for(var i=0;i<SEGS.length;i++){
    var seg=SEGS[i];
    var a0=ang+i*sa-Math.PI/2, a1=a0+sa;
    ctx.beginPath(); ctx.moveTo(cx,cy); ctx.arc(cx,cy,r,a0,a1); ctx.closePath();
    ctx.fillStyle=seg.c; ctx.fill();
    ctx.strokeStyle='rgba(255,255,255,.3)'; ctx.lineWidth=1.5; ctx.stroke();
    var ma=a0+sa/2, lx=cx+(r*.62)*Math.cos(ma), ly=cy+(r*.62)*Math.sin(ma);
    ctx.save(); ctx.translate(lx,ly); ctx.rotate(ma+Math.PI/2);
    ctx.fillStyle='#fff'; ctx.font='bold '+(W*.065)+'px Nunito';
    ctx.textAlign='center'; ctx.textBaseline='middle';
    ctx.fillText('\u20b9'+seg.l,0,0); ctx.restore();
  }
  ctx.beginPath(); ctx.arc(cx,cy,r,0,Math.PI*2);
  ctx.strokeStyle='#f5c518'; ctx.lineWidth=7; ctx.stroke();
  for(var i=0;i<18;i++){
    var a=(i/18)*Math.PI*2, bx=cx+(r+1)*Math.cos(a), by=cy+(r+1)*Math.sin(a);
    ctx.beginPath(); ctx.arc(bx,by,4.5,0,Math.PI*2);
    ctx.fillStyle=i%2===0?'#fff':'#f5c518'; ctx.fill();
  }
}

function doSpin(){
  var today=new Date().toDateString();
  if(S.lastSpin===today){ toast('Already spun today! Come back tomorrow.'); return; }
  if(spinning) return;
  spinning=true;
  var winV=Math.random()<.5?1:11;
  var tSeg=SEGS.findIndex(function(s){ return s.v===winV; });
  var sa=(Math.PI*2)/SEGS.length;
  var tA=-(tSeg*sa+sa/2);
  var full=(5+Math.floor(Math.random()*3))*(Math.PI*2);
  var finalA=spinA+full+(tA-spinA%(Math.PI*2));
  var dur=4200, st=performance.now(), sa0=spinA;
  function ease(t){ return 1-Math.pow(1-t,4); }
  function anim(now){
    var t=Math.min((now-st)/dur,1);
    spinA=sa0+(finalA-sa0)*ease(t);
    drawWheel(spinA);
    if(t<1){ requestAnimationFrame(anim); }
    else {
      spinning=false; S.lastSpin=today; S.balance+=winV; S.totalEarned+=winV;
      S.missions.m3=true; addTxn('Spin Win',winV,'in'); save(); updBals(); updSpinUI();
      snd('coin'); toast('You won '+winV+'!');
    }
  }
  requestAnimationFrame(anim);
}
